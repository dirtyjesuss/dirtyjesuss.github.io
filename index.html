original regex

^(?=.*\d)(?=.*[a-z])(?=.*[A-Z]).{8,}$

pseudo code

[
regexToken.startLine,
regexToken.assertSubPattternWithoutConsuming([
	regexToken.anyCharacter(regexQuantifier.zeroOrMore),
	regexToken.digit
]),
regexToken.assertSubPattternWithoutConsuming([
	regexToken.anyCharacter(regexQuantifier.zeroOrMore),
	regexToken.characterInRange('a', 'z')
]),
regexToken.assertSubPattternWithoutConsuming([
	regexToken.anyCharacter(regexQuantifier.zeroOrMore),
	regexToken.characterInRange('A', 'Z')
]),
regexToken.anyCharacter(regexQuantifier.exactNumberOrMore(8)),
regexToken.endLine
]
